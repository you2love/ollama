import ollama

class ChatBot:
    def __init__(self):
        self.system_prompt = """你是一个客服助手，负责回答用户关于产品的问题。
        请保持友好、专业，使用户满意。"""
        self.conversation = [{'role': 'system', 'content': self.system_prompt}]
    
    def chat(self, user_input):
        self.conversation.append({'role': 'user', 'content': user_input})
        
        response = ollama.chat(
            model='qwen2.5',
            messages=self.conversation
        )
        
        reply = response['message']['content']
        self.conversation.append({'role': 'assistant', 'content': reply})
        return reply
    
    def reset(self):
        self.conversation = [{'role': 'system', 'content': self.system_prompt}]

# 使用
bot = ChatBot()
while True:
    user_input = input("你: ")
    if user_input.lower() in ['exit', 'quit']:
        break
    print("助手:", bot.chat(user_input))

🚀 场景四：批量文档处理

批量摘要生成

import ollama
from concurrent.futures import ThreadPoolExecutor

def summarize_document(text):
    prompt = f"请用 100 字以内概括以下内容：\n\n{text[:1000]}"
    response = ollama.chat(
        model='llama3.2',
        messages=[{'role': 'user', 'content': prompt}]
    )
    return response['message']['content']

def batch_summarize(documents):
    with ThreadPoolExecutor(max_workers=3) as executor:
        summaries = list(executor.map(summarize_document, documents))
    return summaries

# 使用
docs = [doc1, doc2, doc3, ...]
results = batch_summarize(docs)
